# CivicGraph — API Contract

This is the only document the frontend and backend both need to agree on. Once locked, they build in parallel.

## TypeScript types

```typescript
// === Shared types ===

export type EntityType = 'person' | 'org' | 'gov';

export interface PersonSummary {
  id: string;             // hashed canonical ID
  name: string;           // canonical display name
  province?: string;      // 2-letter, e.g. 'AB', 'ON'
  boards: number;         // count of distinct orgs
  totalFunding: number;   // sum of all public $ to those orgs (CAD)
  confidence?: number;    // 0–1, entity resolution confidence
}

export interface GraphNode {
  id: string;
  label: string;          // displayed text
  type: EntityType;
  size?: number;          // for visual scaling (e.g. funding amount)
  province?: string;
  jurisdiction?: string;
}

export interface GraphEdge {
  source: string;         // node id
  target: string;         // node id
  amount?: number;        // for FUNDED or GIFTS_TO edges
  fiscalYear?: number;
  program?: string;
  role?: string;          // for SITS_ON edges (e.g. 'Director', 'Chair')
  yearStart?: number;
  yearEnd?: number;
  edgeType: 'SITS_ON' | 'FUNDED' | 'GIFTS_TO';
  sourceFilingId?: string; // CRA T3010 id, federal grant id, etc.
  sourceUrl?: string;     // direct link to public record
}

export interface ProvenanceRecord {
  id: string;
  type: 'cra_t3010' | 'fed_grant' | 'ab_grant';
  description: string;
  url: string;            // publicly accessible URL to the source filing
  fiscalYear?: number;
  amount?: number;
}

export interface RankedPerson extends PersonSummary {
  compositeScore: number;
}

// === Endpoint shapes ===

export interface TopResponse {
  results: RankedPerson[];
  generatedAt: string;    // ISO8601
}

export interface SearchResponse {
  query: string;
  results: PersonSummary[];
}

export interface PersonDetailResponse {
  person: PersonSummary;
  graph: { nodes: GraphNode[]; edges: GraphEdge[] };
  provenance: ProvenanceRecord[];
  narrative: string;      // Live-generated by Claude Sonnet 4.6 via Bedrock
}
```

## Endpoints

| Method | Path | Response | Notes |
|---|---|---|---|
| GET | `/api/top?n=20` | `TopResponse` | Computed once during ETL, served from S3 cache |
| GET | `/api/search?q={string}` | `SearchResponse` | Up to 20 fuzzy-matched results |
| GET | `/api/person/{id}` | `PersonDetailResponse` | 1-hop ego graph + live Bedrock narrative |
